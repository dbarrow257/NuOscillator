215inline std::vector<FLOAT_T> logspace(FLOAT_T Emin, FLOAT_T Emax, int nDiv) {
216 if (nDiv==0) {
217 std::cerr << "Requested log spacing distribution with 0 divisions" << std::endl;
218 throw std::runtime_error("Invalid setup");
219 }
220
221 std::vector<FLOAT_T> logpoints(nDiv+1, 0.0);
222 logpoints[0]=Emin;
223
224 if (Emin == 0.) {
225 Emin = 0.01;
226 }
227
228 FLOAT_T Emin_log,Emax_log;
229 Emin_log = log10(Emin);
230 Emax_log = log10(Emax);
231
232 FLOAT_T step_log = (Emax_log - Emin_log)/FLOAT_T(nDiv);
233
234 FLOAT_T EE = Emin_log+step_log;
235
236 for (int i=1; i<nDiv; i++) {
237 logpoints[i] = pow(10.,EE);
238 EE += step_log;
239 }
240
241 logpoints[nDiv]=Emax;
242
243 return logpoints;
244}

246inline std::vector<FLOAT_T> linspace(FLOAT_T Emin, FLOAT_T Emax, int nDiv) {
247 if (nDiv==0) {
248 std::cerr << "Requested linear spacing distribution with 0 divisions" << std::endl;
249 throw std::runtime_error("Invalid setup");
250 }
251
252 std::vector<FLOAT_T> linpoints(nDiv+1, 0.0);
253
254 FLOAT_T step_lin = (Emax - Emin)/FLOAT_T(nDiv);
255
256 FLOAT_T EE = Emin;
257
258 for (int i=0; i<nDiv; i++) {
259 if (fabs(EE)<1e-6) {EE = 0.;}
260
261 linpoints[i] = EE;
262 EE += step_lin;
263 }
264
265 linpoints[nDiv] = Emax;
266
267 return linpoints;
