242inline std::vector<FLOAT_T> logspace(FLOAT_T Emin, FLOAT_T Emax, int nDiv) {
243 if (nDiv==0) {
244 std::cerr << "Requested log spacing distribution with 0 divisions" << std::endl;
245 throw std::runtime_error("Invalid setup");
246 }
247
248 std::vector<FLOAT_T> logpoints(nDiv+1, 0.0);
249 logpoints[0]=Emin;
250
251 if (Emin == 0.) {
252 Emin = 0.01;
253 }
254
255 FLOAT_T Emin_log,Emax_log;
256 Emin_log = log10(Emin);
257 Emax_log = log10(Emax);
258
259 FLOAT_T step_log = (Emax_log - Emin_log)/FLOAT_T(nDiv);
260
261 FLOAT_T EE = Emin_log+step_log;
262
263 for (int i=1; i<nDiv; i++) {
264 logpoints[i] = pow(10.,EE);
265 EE += step_log;
266 }
267
268 logpoints[nDiv]=Emax;
269
270 return logpoints;
271}

281inline std::vector<FLOAT_T> linspace(FLOAT_T Emin, FLOAT_T Emax, int nDiv) {
282 if (nDiv==0) {
283 std::cerr << "Requested linear spacing distribution with 0 divisions" << std::endl;
284 throw std::runtime_error("Invalid setup");
285 }
286
287 std::vector<FLOAT_T> linpoints(nDiv+1, 0.0);
288
289 FLOAT_T step_lin = (Emax - Emin)/FLOAT_T(nDiv);
290
291 FLOAT_T EE = Emin;
292
293 for (int i=0; i<nDiv; i++) {
294 if (fabs(EE)<1e-6) {EE = 0.;}
295
296 linpoints[i] = EE;
297 EE += step_lin;
298 }
299
300 linpoints[nDiv] = Emax;
301
302 return linpoints;
303}

313inline std::vector<FLOAT_T> ReadBinEdgesFromFile(std::string TFileName, std::string HistogramName, int Verbose=NuOscillator::Verbosity::NONE) {
314 std::vector<FLOAT_T> BinEdges;
315
316 TFile* File = new TFile(TFileName.c_str());
317 if (!File || File->IsZombie()) {
318 std::cerr << "Could not find file:" << TFileName << std::endl;
319 throw std::runtime_error("Invalid input file.");
320 }
321
322 TH1* Histogram = (TH1*)File->Get(HistogramName.c_str());
323 if (!Histogram) {
324 std::cerr << "Could not find Histogram:" << HistogramName << " in File:" << TFileName << std::endl;
325 throw std::runtime_error("Invalid input file.");
326 }
327
328 BinEdges.resize(Histogram->GetNbinsX()+1);
329 for (int iBin=0;iBin<=Histogram->GetNbinsX();iBin++) {
330 BinEdges[iBin] = Histogram->GetBinLowEdge(iBin+1);
331 }
332
333 delete Histogram;
334 delete File;
335
336 if (Verbose >= NuOscillator::INFO) {
337 std::cout << "Bin edges successfully read from File:" << TFileName << " , Histogram:" << HistogramName << " :=" << std::endl;
338 for (size_t i=0;i<BinEdges.size();i++) {
339 std::cout << BinEdges[i] << ", ";
340 }
341 std::cout << std::endl;
342 }
343
344 return BinEdges;
345}
