185inline NuOscillator::OscillationChannel ReturnOscillationChannel(std::string InputString) {
186 std::string GeneratedFlavour = "";
187 std::string DetectedFlavour = "";
188
189 size_t Delimiter = InputString.find(":");
190 if (Delimiter != std::string::npos) {
191 GeneratedFlavour = InputString.substr(0,Delimiter);
192 DetectedFlavour = InputString.substr(Delimiter+1,InputString.size());
193 } else {

202inline std::vector<FLOAT_T> logspace(FLOAT_T Emin, FLOAT_T Emax, int nDiv) {
203 if (nDiv==0) {
204 std::cerr << "Requested log spacing distribution with 0 divisions" << std::endl;
205 throw;
206 }
207
208 std::vector<FLOAT_T> logpoints(nDiv+1, 0.0);
209 logpoints[0]=Emin;
210
211 if (Emin == 0.) {
212 Emin = 0.01;
213 }
214
215 FLOAT_T Emin_log,Emax_log;
216 Emin_log = log10(Emin);
217 Emax_log = log10(Emax);
218
219 FLOAT_T step_log = (Emax_log - Emin_log)/FLOAT_T(nDiv);
220
221 FLOAT_T EE = Emin_log+step_log;
222
223 for (int i=1; i<nDiv; i++) {
224 logpoints[i] = pow(10.,EE);
225 EE += step_log;
226 }
227
228 logpoints[nDiv]=Emax;
229
230 return logpoints;
231}

233inline std::vector<FLOAT_T> linspace(FLOAT_T Emin, FLOAT_T Emax, int nDiv) {
234 if (nDiv==0) {
235 std::cerr << "Requested linear spacing distribution with 0 divisions" << std::endl;
236 throw;
237 }
238
239 std::vector<FLOAT_T> linpoints(nDiv+1, 0.0);
240
241 FLOAT_T step_lin = (Emax - Emin)/FLOAT_T(nDiv);
242
243 FLOAT_T EE = Emin;
244
245 for (int i=0; i<nDiv; i++) {
246 if (fabs(EE)<1e-6) {EE = 0.;}
247
248 linpoints[i] = EE;
249 EE += step_lin;
250 }
251
252 linpoints[nDiv] = Emax;
253
254 return linpoints;
255}
