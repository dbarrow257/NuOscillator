179inline int NeutrinoFlavour_StrToInt(const std::string& NuFlav) {

282inline std::vector<FLOAT_T> logspace(FLOAT_T Emin, FLOAT_T Emax, int nDiv) {
283 if (nDiv==0) {
284 std::cerr << "Requested log spacing distribution with 0 divisions" << std::endl;
285 throw std::runtime_error("Invalid setup");
286 }
287
288 std::vector<FLOAT_T> logpoints(nDiv+1, 0.0);
289 logpoints[0]=Emin;
290
291 if (Emin == 0.) {
292 Emin = 0.01;
293 }
294
295 FLOAT_T Emin_log,Emax_log;
296 Emin_log = log10(Emin);
297 Emax_log = log10(Emax);
298
299 FLOAT_T step_log = (Emax_log - Emin_log)/FLOAT_T(nDiv);
300
301 FLOAT_T EE = Emin_log+step_log;
302
303 for (int i=1; i<nDiv; i++) {
304 logpoints[i] = pow(10.,EE);
305 EE += step_log;
306 }
307
308 logpoints[nDiv]=Emax;
309
310 return logpoints;
311}

321inline std::vector<FLOAT_T> linspace(FLOAT_T Emin, FLOAT_T Emax, int nDiv) {
322 if (nDiv==0) {
323 std::cerr << "Requested linear spacing distribution with 0 divisions" << std::endl;
324 throw std::runtime_error("Invalid setup");
325 }
326
327 std::vector<FLOAT_T> linpoints(nDiv+1, 0.0);
328
329 FLOAT_T step_lin = (Emax - Emin)/FLOAT_T(nDiv);
330
331 FLOAT_T EE = Emin;
332
333 for (int i=0; i<nDiv; i++) {
334 if (fabs(EE)<1e-6) {EE = 0.;}
335
336 linpoints[i] = EE;
337 EE += step_lin;
338 }
339
340 linpoints[nDiv] = Emax;
341
342 return linpoints;
343}

353inline std::vector<FLOAT_T> ReadBinEdgesFromFile(std::string TFileName, std::string HistogramName, int Verbose=NuOscillator::Verbosity::NONE) {
354 std::vector<FLOAT_T> BinEdges;
355
356 TFile* File = new TFile(TFileName.c_str());
357 if (!File || File->IsZombie()) {
358 std::cerr << "Could not find file:" << TFileName << std::endl;
359 throw std::runtime_error("Invalid input file.");
360 }
361
362 TH1* Histogram = (TH1*)File->Get(HistogramName.c_str());
363 if (!Histogram) {
364 std::cerr << "Could not find Histogram:" << HistogramName << " in File:" << TFileName << std::endl;
365 throw std::runtime_error("Invalid input file.");
366 }
367
368 BinEdges.resize(Histogram->GetNbinsX()+1);
369 for (int iBin=0;iBin<=Histogram->GetNbinsX();iBin++) {
370 BinEdges[iBin] = Histogram->GetBinLowEdge(iBin+1);
371 }
372
373 File->Close();
374 delete File;
375
376 if (Verbose >= NuOscillator::INFO) {
377 std::cout << "Bin edges successfully read from File:" << TFileName << " , Histogram:" << HistogramName << " :=" << std::endl;
378 for (size_t i=0;i<BinEdges.size();i++) {
379 std::cout << BinEdges[i] << ", ";
380 }
381 std::cout << std::endl;
382 }
383
384 return BinEdges;
385}
